```bash
# Download, unzip, compile and install python3.10.8
cd Downloads/
wget https://www.python.org/ftp/python/3.10.8/Python-3.10.8.tgz
tar xzf Python-3.10.8.tgz
cd Python-3.10.8
./configure --enable-optimizations
make altinstall
python3.10 -V
```
