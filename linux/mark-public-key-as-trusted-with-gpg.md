# Import key
gpg --import <keyfile>
# Find fingerprint of key
gpg --fingerprint
# Open GPG interactive prompt
gpg --edit-key <fingerprint>
# Select trust level and confirm by typing `y` (4 for full trust, 5 for ultimate trust)
trust
